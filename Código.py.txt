#IMC => PESO (KG) / ALTURA (M)²

def numero_quadrado(numero): 
  quadrado = numero * numero
  return quadrado

def calcular_imc(peso, altura): 
  altura_quadrada = numero_quadrado(altura)
  meu_imc = peso / altura_quadrada    
  return meu_imc
